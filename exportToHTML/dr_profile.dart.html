import 'package:flutter/material.dart';
import 'package:untitled2/ductur/statistics.dart';
import 'package:untitled2/ductur/all_reports.dart';
import 'package:untitled2/ductur/payments.dart';
import 'package:untitled2/ductur/history.dart';

class DrProfilePage extends StatelessWidget {
  final Map<String, dynamic> doctorData;

  const DrProfilePage({Key? key, required this.doctorData}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text("Doctor Profile"), backgroundColor: Colors.blueAccent),
      body: SingleChildScrollView(
        padding: const EdgeInsets.all(16.0),
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            // Profile Section
            Center(
              child: Column(
                children: [
                  CircleAvatar(
                    radius: 50,
                    backgroundImage: AssetImage('assets/images/doctor_placeholder.png'),
                  ),
                  SizedBox(height: 10),
                  Text(doctorData['name'], style: _headerStyle),
                  Text("Specialisation: ${doctorData['specialisation']}", style: _subTextStyle),
                  SizedBox(height: 5),
                  Text("Doctor ID: D-${doctorData.hashCode % 10000}", style: _idStyle),
                ],
              ),
            ),
            SizedBox(height: 20),

            // Hospital and Charges
            _buildCard(
              title: "Hospital: ${doctorData['hospital']}",
              subtitle: "Consultation Charges: ‚Çπ1500",
            ),

            SizedBox(height: 20),

            // Current Appointments
            _buildContainer(
              icon: Icons.local_hospital,
              text: "Currently Appointed: ${doctorData['patientsAssigned'].length} patients",
            ),

            SizedBox(height: 20),

            // Summary
            _buildCard(
              title: "Doctor Summary",
              subtitle: "Experience: ${doctorData['experience']} years\n"
                  "Revenue Earned: ‚Çπ${doctorData['revenueEarned']}\n"
                  "Successful Treatments: 120+",
            ),

            SizedBox(height: 20),

            // Footer Buttons
            Column(
              children: [
                _buildFooterButton(context, Icons.bar_chart, "Statistics", StatisticsPage()),
                _buildFooterButton(context, Icons.picture_as_pdf, "All Reports", AllReportsPage()),
                _buildFooterButton(context, Icons.payment, "Payments", PaymentsPage()),
                _buildFooterButton(context, Icons.history, "History", HistoryPage()),
              ],
            ),
          ],
        ),
      ),
    );
  }

  // üè• Reusable Card Widget
  Widget _buildCard({required String title, required String subtitle}) {
    return Card(
      shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(10)),
      elevation: 4,
      child: Padding(
        padding: const EdgeInsets.all(16.0),
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            Text(title, style: _sectionTitleStyle),
            SizedBox(height: 8),
            Text(subtitle, style: _sectionTextStyle),
          ],
        ),
      ),
    );
  }

  // üîπ Reusable Info Container
  Widget _buildContainer({required IconData icon, required String text}) {
    return Container(
      padding: EdgeInsets.all(16),
      decoration: BoxDecoration(
        color: Colors.blue[50],
        borderRadius: BorderRadius.circular(10),
      ),
      child: Row(
        children: [
          Icon(icon, color: Colors.blue, size: 40),
          SizedBox(width: 10),
          Expanded(
            child: Text(text, style: TextStyle(fontSize: 16, fontWeight: FontWeight.bold)),
          ),
        ],
      ),
    );
  }

  // üîπ Footer Button Widget
  Widget _buildFooterButton(BuildContext context, IconData icon, String title, Widget page) {
    return Card(
      elevation: 3,
      margin: EdgeInsets.symmetric(vertical: 8),
      child: ListTile(
        leading: Icon(icon, color: Colors.blue),
        title: Text(title, style: TextStyle(fontWeight: FontWeight.bold)),
        trailing: Icon(Icons.arrow_forward_ios, size: 18),
        onTap: () => Navigator.push(context, MaterialPageRoute(builder: (context) => page)),
      ),
    );
  }

  // üé® Styles
  final TextStyle _headerStyle = const TextStyle(fontSize: 22, fontWeight: FontWeight.bold);
  final TextStyle _subTextStyle = const TextStyle(fontSize: 16, color: Colors.grey);
  final TextStyle _idStyle = const TextStyle(fontSize: 14, color: Colors.grey);
  final TextStyle _sectionTitleStyle = const TextStyle(fontSize: 18, fontWeight: FontWeight.bold);
  final TextStyle _sectionTextStyle = const TextStyle(fontSize: 16, color: Colors.grey);
}
